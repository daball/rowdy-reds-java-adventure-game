public void fightDragon() {
  while (dragon.isAlive()) {
    if (me.getHealthPoints() <= 30) {
    me.heal(salve);
    }
    else if (dragon.isInhaling()) {
    me.block(shield);
    }
    else if (dragon.isFlying()) {
    me.attack(dragon, crossbow);
    }
    else {
    me.attack(dragon, sword);
    }
  }
}
